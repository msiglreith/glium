Example

let bindings = vec![(
        format!("position"), 0,
        glium::vertex::AttributeType::F32F32,
    ), (
        format!("color"), 2 * ::std::mem::size_of::<f32>(),
        glium::vertex::AttributeType::F32,
    ),
];

    1.0, -0.3, 409.0,
    -0.4, 2.8, 715.0f32
];

    glium::vertex::PerInstanceAttributesBuffer::new_raw(&display, data, bindings, 3 * ::std::mem::size_of::<f32>())
};
